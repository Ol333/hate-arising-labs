import cv2

# –ü—É—Ç—å –∫ –æ–±—É—á–µ–Ω–Ω–æ–π –º–æ–¥–µ–ª–∏
model = YOLO("/content/runs/detect/train/weights/best.pt")

fps = cap.get(cv2.CAP_PROP_FPS)

height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))

out = cv2.VideoWriter(output_video_path, fourcc, fps, (width, height))

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # –°–∂–∏–º–∞–µ–º –∫–∞–¥—Ä –¥–æ 640x640
    resized_frame = cv2.resize(frame, (640, 640))

    # –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ–º –∫–∞–¥—Ä –º–æ–¥–µ–ª—å—é

    # –ú–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –æ–±—Ä–∞—Ç–Ω–æ –≤ 1080x1920
    for result in results:
        for box in result.boxes.xyxy:
            x1, y1, x2, y2 = box[:4]
            x1 = int(x1 * (width / 640))
            x2 = int(x2 * (width / 640))
            y1 = int(y1 * (height / 640))
            y2 = int(y2 * (height / 640))

            # –†–∏—Å—É–µ–º –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫ –Ω–∞ –∏—Å—Ö–æ–¥–Ω–æ–º –∫–∞–¥—Ä–µ
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)

    # –°–æ—Ö—Ä–∞–Ω—è–µ–º –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–π –∫–∞–¥—Ä
    out.write(frame)

cap.release()
out.release()
print("–û–±—Ä–∞–±–æ—Ç–∫–∞ –≤–∏–¥–µ–æ –∑–∞–≤–µ—Ä—à–µ–Ω–∞.")

from deep_sort_realtime.deepsort_tracker import DeepSort

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è YOLO –º–æ–¥–µ–ª–∏
model_path = "/content/runs/detect/train2/weights/best.pt"  # –£–∫–∞–∂–∏—Ç–µ –ø—É—Ç—å –∫ –≤–∞—à–µ–π –º–æ–¥–µ–ª–∏
model = YOLO(model_path)

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Ç—Ä–µ–∫–µ—Ä–∞ DeepSORT
tracker = DeepSort(max_age=70, n_init=5, max_cosine_distance=0.7, nn_budget=None)

# –û—Ç–∫—Ä—ã–≤–∞–µ–º –≤–∏–¥–µ–æ
video_path = "/content/dcf73189-VID_20230408_143121.mp4"
output_path = "output_video_3.mp4"
cap = cv2.VideoCapture(video_path)

# –ü–æ–ª—É—á–∞–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–∏ –∏ FPS –≤–∏–¥–µ–æ
fps = int(cap.get(cv2.CAP_PROP_FPS))
frame_width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
frame_height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
fourcc = cv2.VideoWriter_fourcc(*'mp4v')
out = cv2.VideoWriter(output_path, fourcc, fps, (frame_width, frame_height))

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # –°–∂–∏–º–∞–µ–º –∫–∞–¥—Ä –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ YOLO
    resized_frame = cv2.resize(frame, (640, 640))

    # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–∞–¥—Ä–∞ –º–æ–¥–µ–ª—å—é YOLO
    results = model(resized_frame, conf=0.6, iou=0.5)

    detections = []
    for result in results:
        if len(result.boxes.xyxy) > 0:
            # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º t–µ–Ω–∑–æ—Ä –≤ numpy –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞
            boxes = result.boxes.xyxy.cpu().numpy()  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –≤ numpy –º–∞—Å—Å–∏–≤
            conf = float(result.boxes.conf.cpu().numpy()[0])  # –£–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å –º–æ–¥–µ–ª–∏
            cls = int(result.boxes.cls.cpu().numpy()[0])  # –ö–ª–∞—Å—Å –æ–±—ä–µ–∫—Ç–∞
            for box in boxes:
                x1, y1, x2, y2 = box[:4]
                # –ú–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –æ–±—Ä–∞—Ç–Ω–æ –≤ –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω–æ–µ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–µ
                x1 = int(x1 * (frame_width / 640))
                x2 = int(x2 * (frame_width / 640))
                y1 = int(y1 * (frame_height / 640))
                y2 = int(y2 * (frame_height / 640))
                # –§–æ—Ä–º–∞—Ç –¥–ª—è —Ç—Ä–µ–∫–µ—Ä–∞: [x1, y1, width, height, confidence, class_id]
                detections.append([[x1, y1, x2 - x1, y2 - y1], conf, cls])

    # –û–±–Ω–æ–≤–ª—è–µ–º —Ç—Ä–µ–∫–µ—Ä
    tracks = tracker.update_tracks(detections, frame=frame)

    # –û—Ç–æ–±—Ä–∞–∂–∞–µ–º —Ç—Ä–µ–∫–∏ –Ω–∞ –∫–∞–¥—Ä–µ
    for track in tracks:
        if not track.is_confirmed():
            continue

        track_id = track.track_id
        ltrb = track.to_ltrb()
        x1, y1, x2, y2 = map(int, ltrb)

        # –†–∏—Å—É–µ–º –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫ –∏ ID —Ç—Ä–µ–∫–∞
        cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
        label = f"ID {track_id}"
        cv2.putText(frame, label, (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

    # –°–æ—Ö—Ä–∞–Ω—è–µ–º –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã–π –∫–∞–¥—Ä
    out.write(frame)

cap.release()
out.release()
print(f"–û–±—Ä–∞–±–æ—Ç–∫–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞, —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Å–æ—Ö—Ä–∞–Ω—ë–Ω –≤ {output_path}")
